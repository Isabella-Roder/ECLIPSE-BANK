    resumo mensal

    dashboard {
        visao geral do sistema
        saldo total
        total de contas
        total de usuarios
        ultimas transacoes
    }

    usuarios {
        cadastro de usuarios
        nome

        cpf

    }

    contas {

        listar contas
        deposito
        saque
        saldo
        limite
        chave pix
    }

    transferencias {
        conta origem
        conta destino
        valor
        registrar no extrato das duas contas
    }

    extrato {
        buscar por conta
        listar transacoes
        filtrar por tipo
        filtrar por data
        filtrar por categoria

    dados {

        mostrar dados pessoais
        mostrar nome social quando existir

        mostrar saldo

        chave pix
        limite
        atalhos para transferencia e extrato
    }
}
